program sumaPares;
var suma = 0;
var temp = 0;
var n = 2;
temp = n;
var m = 10;

if (n > m) {
    print("El primer número debe ser menor que el segundo.");
} else {
    while (n <= m) {
        if (n % 2 == 0) {
            suma = suma + n;
        }
        n = n + 1;
    }
    print("La suma de los números pares entre " + temp + " y " + m + " es " + suma + ".");
}
